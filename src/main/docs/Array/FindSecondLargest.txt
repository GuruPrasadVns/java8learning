public int findSecondLargest(int[] arr){
    int n = arr.length;
    int max = Integer.MIN_VALUE;
    int secondMax = Integer.MIN_VALUE;

    for(int i = 0 ; i < n; i++){
        if(arr[i] != max && arr[i] != secondMax){
            if(arr[i]) > max){
                secondMax = max;
                max = arr[i];
            }
            if(arr[i]) > secondMax){
                secondMax = arr[i];
            }
        }
    }

    return secondMax;
}
