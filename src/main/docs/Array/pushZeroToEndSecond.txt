public static void pushZeroToEndSecond(int[] arr){
    int n = arr.length;
    int j = 0;

    for(int i = 0; i < n; i++){
        if(arr[i] != 0){
            swap(arr, i , j++);
        }
    }
}

n = 8;
j = 0;

for loop
i = 0; i < 8 (true)
-----------------------------------------------------------------------------------------------------------------------
    if(arr[i] != 0) => if(arr[0] != 0) => if(1 != 0) : true
    -------------------------------------------------------------------------------------------------------
        swap(arr, i, j++) => swap(arr, 0, 0++) => arr is not changed and j = 1

i = 1; i < 8 (true)
-----------------------------------------------------------------------------------------------------------------------
    if(arr[i] != 0) => if(arr[1] != 0) => if(2 != 0) : true
    -------------------------------------------------------------------------------------------------------
        swap(arr, i, j++) => swap(arr, 1, 1++) => arr is not changed and j = 2
